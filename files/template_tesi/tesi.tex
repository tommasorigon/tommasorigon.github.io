\chapter{Esempi}

% --- ESEMPIO TEOREMA ---
\begin{theorem}[Teorema di Pitagora]
Dato un triangolo rettangolo di lati $a, b$ e $c$, dove $c$ è l'ipotenusa e $a$ e $b$ sono i cateti, allora
$$
    a^2 + b^2 = c^2.
$$
\end{theorem}

% --- ESEMPIO FORMULA MATEMATICA ---
\subsubsection{Esempio formula matematica}

Una semplice equazione numerata:
\begin{equation}
    3^2 + 4^2 = 5^2
\end{equation}
Per formule NON numerate, usare \texttt{\$\$ ... \$\$} o \texttt{equation*}:
$$
    e = mc^2
$$

% --- ESEMPIO MATRICE ---
\subsubsection{Esempio matrice}

Una matrice Jacobiana con notazione matriciale:
\begin{equation}
    j(\tau,\zeta) = \begin{bmatrix}
        j_{\zeta \zeta}(\tau,\zeta) & j_{\zeta \tau}(\tau,\zeta) \\
        j_{\tau \zeta}(\tau,\zeta) & j_{\tau \tau}(\tau,\zeta)
    \end{bmatrix}
\end{equation}

% --- ESEMPIO CITAZIONE ---
\subsubsection{Esempio citazione}

Per una citazione completa: \citet{pacesalvan}. Per una citazione tra parentesi: \citep{pacesalvan}.
